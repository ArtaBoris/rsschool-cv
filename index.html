            function tickets(peopleInLine){
                const moneys = {
                    25: 0,
                    50: 0,
                    100: 0
                }
                for (let i = 0; i < peopleInLine.length; i++){
                    const money = peopleInLine[i]
                    if (money === 25) {
                        moneys[money]++
                    }
                    if (money === 50) {
                        if (moneys[25] > 0) {
                            moneys[25]--
                            moneys[money]++
                        }
                        else return 'NO'
                    }
                    if (money === 100) {
                        if (moneys[50] > 0 && moneys[25] > 0) {
                            moneys[50]--
                            moneys[25]--
                            moneys[money]++
                        }
                        else if (moneys[25] >=3) {
                            moneys[25] -= 3
                            moneys[money]++
                        }
                        else return 'NO'
                        }  
                    }
                return 'YES'  
                }
